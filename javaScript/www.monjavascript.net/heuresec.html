Indiquez l'heure sur la barre d'état
 à la seconde près

Regarder la barre d'état

L'heure indiquer est celle de l'ordinateur du visiteur. C'est mieux si il est à l'heure

Insérez le code ci-dessous dans votre page d'accueil


<script language="JavaScript">
// PLF - http://www.monjavascript.net/
var date_du_jour, heure, minute, mess;
function heuremess() {
date_du_jour = new Date();
heure = date_du_jour.getHours();
minute = date_du_jour.getMinutes();
seconde = date_du_jour.getSeconds();
// Test pour determiner l'affichage des heures
if (heure<10)
heure="0"+ heure+ "H";
else
heure= heure + "H";
// Test pour determiner l'affichage des minutes
if (minute<10)
minute="0" + minute;
else
minute=minute;
// Test pour determiner l'affichage des secondes
if (seconde<10)
seconde =":0" + seconde;
else
seconde =":" + seconde;
window.status = heure + minute + seconde;
window.setTimeout("heuremess()",1000);
}
heuremess();
</script>

<body bgcolor="#FFFFFF" background="im/fond.gif" onUnload="window.status = '';">
