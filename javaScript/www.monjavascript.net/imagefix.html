Une Image Qui Revient Toujours à La Même Place
Changer la taille de la fenêtre, déplacer vous avec les barre de défilement : l'image au milieu à droite revient toujours à sa place. Ce n'est pas une Popup mais vous pouvez l'utiliser pour un lien publicitaire

Testé avec IE Netscape7 Firefox

Insérez le code ci-dessous dans l'entête de votre page (entre les balises<head> ... </head>)

<script language="JavaScript">
<!--
//PLF-http://www.monjavascript.net/
var hauteur=0;
var temps=30;
var max=30;
function position(left,top,width,height,contenu) {
hauteur=height;
var Z="<DIV id='CalqueImage' style='position:absolute;left:"+left+"; top:"+top+";width:"+width+"; height:"+height+"'>"+contenu+"</DIV>";
document.write(Z);
setTimeout("Start()",temps);
}
function Start() {
DeplacerImage("CalqueImage",Postionimage("CalqueImage"));
setTimeout("Start()",temps);
}
function DeplacerImage(nom,top) {
document.getElementById(nom).style.top = top;
}
function Postionimage(nom) {
var pos=parseInt(document.getElementById(nom).style.top);
var wintop=parseInt(document.body.scrollTop);
var dispo=document.body.clientHeight;
var diff=Math.ceil((pos-(wintop+dispo-hauteur)));
if (diff>max) {diff=max;}
if (diff<-1*max) {diff=-1*max;}
var top=pos-diff;
return top;
}
//-->
</script>

 

Insérez le code ci-dessous dans votre page d'accueil après la balise de fin d'entête </head>)

<script language="JavaScript">
<!--
//PLF-http://www.monjavascript.net/
//position(marge,1,largeur,hauteur(>H image),'image');
position(10,1,100,120,'<img src="monimage.gif" width="88" height="32">');
//-->
</script>

Pour créer uin lien sur l'image remplacer <img src="monimage.gif" width="88" height="32"> par <a href="monlien.htm" target="_blank"><img src="monimage.gif" width="88" height="32" border="0" alt="mon message"></a>
Taille de l'image
